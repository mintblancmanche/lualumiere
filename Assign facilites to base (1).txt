---- Preference section
local side1 = "Argentina"
local unitname = "Port Stanley Airfield"
local unit = ScenEdit_GetUnit({side=side1, name=unitname})

ScenEdit_SetKeyValue("side2", side2)

local unitlist =table.concat(unit.group.unitlist, ', ' )
ScenEdit_SetKeyValue("unitlist", unitlist )
print(ScenEdit_GetKeyValue("unitlist") )


table.remove(unitlist, 1) --grouplead


local facilities = table.concat(unitlist, ', ' )
ScenEdit_SetKeyValue( "facities", facilities )

---- group lead as 
local lead = ScenEdit_GetUnit({guid=unit.group.unitlist[1]})

lead.group = newname
local newbase = ScenEdit_GetUnit({side=side2, name=unitname})

-- new group
ScenEdit_SetKeyValue( "newbase_guid", newbase.guid )
