function VLFcheck(guid)
  local guid = '4f8eae13-54ac-4abb-ab40-b3af5c5fb3b6'
  local unit = ScenEdit_GetUnit({guid=guid})

  if unit.altitude >= -41 then
    ScenEdit_SetKeyValue("VLF", true)
  else
    ScenEdit_SetKeyValue("VLF", false)
  end
  
  return ScenEdit_GetKeyValue("VLF")
end


local guid = '4f8eae13-54ac-4abb-ab40-b3af5c5fb3b6' --Swolf
local unit = ScenEdit_GetUnit({guid=guid})

if unit.altitude >= -41 then
  return true
  ScenEdit_EventX().isActive = false
else
  return false
end


ScenEdit_SetKeyValue("UHF", true)

local side1 = "GREEN"
local side2 = "BLUE"
local posture = "F"
ScenEdit_SetSidePosture(side1, side2, posture)
ScenEdit_MsgBox('Conn, Radio. UHF Datalink is established.', 0)

local event = "Datalink_Deactive"
