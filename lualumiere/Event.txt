◇UnitBaseStatusトリガーを編集するやつ

local trigger = "Trigger1"
local type = "UnitBaseStatus"
local TargetFilter={
TargetSide= "United Kingdom",
TargetType= "Ship",
TargetSubType= nil, -- Fighter
SpecificUnitClass= nil, --MiG-25P
SpecificUnitID= nil, --530 IAP #31 Red (Belenko)
}
local TargetCondition= 0 -- Taxiing to parking spot
local trigger = ScenEdit_SetTrigger({mode=mode, name=trigger, type=type,
TargetFilter=TargetFilter, TargetCondition=TargetCondition})
print(trigger)

local trigger = ScenEdit_SetTrigger({mode="update", name="Blue_AircraftLands", type="UnitBaseStatus"})
print(trigger)
